def balancesheet(code, startyear, endyear):
    period = yearls(startyear, endyear)
    demo = pro.balancesheet(ts_code=code, period=period[0])
    df = pd.DataFrame(columns=demo.columns)
    f = open('balancesheet.pkl', 'rb')
    dic = pickle.load(f)
    column = list()
    for col in df.columns:
        column.append(dic[col])
    for year in period:
        bs = pro.balancesheet(ts_code=code, period=year)
        df = pd.concat([df, bs], ignore_index=True)
    df.columns = column
    for col in df.columns:
        df[col].fillna(np.nan, inplace=True)
    df = df.dropna(axis=1)
    df['报告期'] = df['报告期'].apply(lambda x:x[:4]+'年')
    df.index = df['报告期']
    return df
